class Post(_Base_):
    """A base class for all geometry objects.

    Args:
        identifier: Text string for a unique object ID. Must be < 100 characters and
            not contain any spaces or special characters.


    """

    __slots__ = (
        "name",
        "flpost_name",
        "active",
        "simple",
        "creation_order",
        "settings",
    )

    def __init__(self, name="default_name"):
        _Base_.__init__(self)  # process the identifier
        self.display_name = "post" + self.identifier
        self.settings = {}
        self.settings.update(POST)
        self.type = "planecut"
        self.string = ""
        self.set("name", name)

    def to_str(self):
        """
        将对象转换为字符串表示形式。

        :return: 对象的字符串表示形式。
        """

        title = "post_load_object {" + self.type
        body = " "
        end = "}"
        keys = self.settings.keys()
        values = self.settings.values()
        for index, key in enumerate(keys):
            body_string = str(values[index])
            body += "-" + key + " " + body_string + " "
        string = title + body + end

        if "'{" in string:
            string.replace("'{", "{")
        if "}'" in string:
            string.replace("}'", "}")
        self.string = string
        return string

    def from_plane(self, plane):
