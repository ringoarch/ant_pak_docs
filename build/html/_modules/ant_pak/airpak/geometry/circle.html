class CircleMixIn(_BaseGeo):
    """
    Circle Mixin class

    properties:
        * identifier    : str
        * display_name : str
        * center   : Point3d
        * radius   : float
        * iradius  : float
        * plane    : int
        * circle   : Circle
        * shape    : str
        * shape_begin : str
        * shape_type  : str
        * current_stype : str
        * geometry  : Circle
    """

    def __init__(self):
        """
        初始化CircleMixIn对象。
        """
        _BaseGeo.__init__(self)
        self.center = None
        self.radius = 0
        self.iradius = 0
        self.current_stype = "circ"
        self.circle = None
        self.shape_begin = self.shape_type + "_shape shape_" + self.current_stype

    def from_circ(self, circle):
        """
        从圆形对象中创建一个CircleMixIn对象。

        :param circle: 要创建的圆形对象。

        :return CircleMixIn: 创建的CircleMixIn对象。
        """
        self.geometry = circle
        self.refresh_point()
        self.center = circle.Center
        self.radius = circle.Radius
        self.circle = circle
        normal = circle.Normal
        if normal.X > 0:
            self.plane = 0
        elif normal.Y > 0:
            self.plane = 1
        elif normal.Z > 0:
            self.plane = 2
        self.geo_to_str()
        return self

    def geo_to_str(self):
        """
        将圆形的几何信息转换为字符串表示形式。

        该方法将圆形的中心点、半径、平面类型等信息转换为字符串，并将这些字符串拼接成一个完整的形状字符串。

        :return str: 表示圆形几何信息的字符串。
        """
        str_geo = circle_to_str(self.circle)
        str_split_flag = flag_to_str("split_flag", self.split_flag)

        str_plane = flag_to_str("plane", self.plane)
        str_radius = flag_to_str("radius", self.radius)
        str_iradius = flag_to_str("iradius", self.iradius)
        flags = [str_split_flag, str_plane, str_radius, str_iradius]

        str_flags = list_str(flags)

        shape_begin = "    shape {}\n".format(self.shape_begin)
        shape_body = "        setval {} {}\n".format(str_geo, str_flags)
        shape_end = "    end shape\n"
        shape = shape_begin + shape_body + shape_end
        self.shape = shape
        return shape

    @classmethod
    def from_shape_str(str_geo, str_plane, str_radius):
        """
        从形状字符串中创建一个圆形对象。

        :param str_geo: 表示圆形中心点的字符串。
        :param str_plane: 表示圆形平面类型的字符串。
        :param str_radius: 表示圆形半径的字符串。

        :return CircleMixIn: 创建的圆形对象。
        """
        circle_object = CircleMixIn()
        circle = str_to_circle(str_geo, str_plane, str_radius)
        circle_object = circle_object.from_circ(circle)
        return circle_object
