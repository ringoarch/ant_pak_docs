    def set_vel_by_vector(self, vector, unit="m/s"):
        v_body_settings = [
            "dir_specf",
            "xvecf",
            "xmagn_vec",
            "free_velz_on",
            "theta_anglef",
            "zanglef",
            "velocity_by",
            "zvecf",
            "free_magnitude",
            "zmagn_vec",
            "alpha_anglef",
            "yvecf",
            "ymagn_vec",
        ]

        self.add_body_settings(v_body_settings)
        self.remove_body_settings(
            [
                "pressure",
                "free_press_static",
                "free_press_on",
            ]
        )

        length = vector.Length

        self.free_magnitude = length
        vector.Unitize()

        x, y, z = round(vector.X, 3), round(vector.Y, 3), round(vector.Z, 3)
        # max_num = max([abs(i) for i in list])

        # if max_num != 0:
        #     x, y, z = [i / max_num for i in list]
        # else:
        #     x, y, z = 0, 0, 0

        self.param_units.update({"free_magnitude": unit})
        self.xvecf = x
        self.yvecf = y
        self.zvecf = z

        self.xanglef = x
        self.yanglef = y
        self.zanglef = z

        if abs(x) > 0:
            self.free_velx_on = 1
            self.xmagn_vec = x
        else:
            self.xmagn_vec = 0

        if abs(y) > 0:
            self.free_vely_on = 1
            self.ymagn_vec = y
        else:
            self.ymagn_vec = 0
        if abs(z) > 0:
            self.free_velz_on = 1
            self.zmagn_vec = z
        else:
            self.zmagn_vec = 0

        return True
