class ThermalMixIn(_BaseProperty):
    """
    Properties:
        * identifier
        * display_name
        * base_point
        * rooms
        * geometry
        * story
        * height
        * repeat
    """

    __slots__ = ("power",)

    def __init__(self):
        """
        初始化Block对象。

        该方法调用父类的初始化方法，并设置一些默认属性。
        """
        _BaseProperty.__init__(self)  # process the identifier
        self.power = 0
        self.all_thermal_type = ""
        self.all_temp = AMBIENT_TEMP
        self.all_hollow_heat_tr = 1
        self.all_ambient = AMBIENT_TEMP
        self.all_heat_area = 0
        self.all_heat_type = "area"
        self.thermal_rtype = "reftemp"
        self.thermal_itemp = AMBIENT_TEMP
        self.thermal_type = "temp"
        self.thermal_heat_area = 0
        self.property_type = "thermal"

    def set_power(self, value, unit="W"):
        v_body_settings = ["power"]
        self.temp = 1
        self.add_body_settings(v_body_settings)
        self.param_units.update({"power": unit})
        self.power = value

    def set_temperature(self, value, unit="C"):
        v_body_settings = ["all_thermal_type", "all_temp"]
        self.add_body_settings(v_body_settings)
        if unit == "C":
            try:
                self.param_units.pop("free_temperature")
            except:
                pass
        else:
            self.param_units.update({"free_temperature": unit})
        self.all_thermal_type = "temp"
        self.all_temp = ABS_TEMP + value
