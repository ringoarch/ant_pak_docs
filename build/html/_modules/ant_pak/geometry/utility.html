def offset_curve(curve, distance):

    offset_crvs = curve.Offset(
        rg.Plane.WorldXY, distance, 0.01, rg.CurveOffsetCornerStyle.Sharp
    )
    # 确保偏移结果是一个有效的曲线列表
    if offset_crvs and len(offset_crvs) > 0:
        return offset_crvs
    else:
        print_time("Error: Offset curve failed.")

def convert_to_curves(objects):
    """
    Convert a list of Polylines, PolyCurves, or PolylineCurves to Curves.

    Args:
    objects (list): A list containing Polylines, PolyCurves, or PolylineCurves.

    Returns:
    list: A list of Curves.
    """
    curves = []
    for obj in objects:
        if isinstance(
            obj, (rg.Polyline, rg.Rectangle3d, rg.PolylineCurve, rg.PolyCurve)
        ):
            curve = obj.ToNurbsCurve()
        else:
            curve = obj  # Assume it is already a curve or an unknown type that can't be converted
        curves.append(curve)
    return curves

def boundary_surfaces(curves):
    curves = item_to_list(curves)
    curves = convert_to_curves(curves)

    boundary_surfaces = rg.Brep.CreatePlanarBreps(curves)
    # 确保创建的边界曲面是有效的
    if boundary_surfaces and len(boundary_surfaces) > 0:
        return boundary_surfaces
    else:
        print_time("Error: Could not create boundary surfaces.")

def extrude_line_to_surface(line, vector):
    """
    将一条线沿一个向量拉伸为一个面
    :param line: 要拉伸的线 (Curve)
    :param vector: 拉伸的向量 (Vector3d)
    :return: 生成的面 (Surface)
    """
    # 检查输入是否为线
    if not isinstance(line, (rg.LineCurve, rg.Line)):
        return None, "Input must be a LineCurve."

    # 创建拉伸
    surface = rg.Surface.CreateExtrusion(line, vector)

    return surface

def move(geometry, translation_vector):
    """
    移动几何体，可以是Point3d, Line或Curve
    :param geometry: 要移动的几何体 (Point3d, Line, Curve)
    :param translation_vector: 移动的向量 (Vector3d)
    :return: 移动后的几何体
    """
    # 创建移动变换
    translation = rg.Transform.Translation(translation_vector)

    if isinstance(geometry, rg.Point3d):
        # 移动Point3d
        moved_geometry = geometry + translation_vector
    elif isinstance(geometry, rg.Line):
        # 移动Line
        moved_geometry = rg.Line(
            geometry.From + translation_vector, geometry.To + translation_vector
        )
    elif isinstance(geometry, (rg.Curve, rg.Brep)):
        # 移动Curve
        moved_geometry = geometry.Duplicate()
        moved_geometry.Transform(translation)
    elif isinstance(geometry, rg.NurbsSurface):
        moved_geometry = geometry.Duplicate()
        control_points = moved_geometry.Points

        for u in range(control_points.CountU):
            for v in range(control_points.CountV):
                control_point = control_points.GetControlPoint(u, v)
                moved_point = rg.ControlPoint(
                    control_point.Location + translation_vector, control_point.Weight
                )
                control_points.SetControlPoint(u, v, moved_point)
    else:
        print_time("Unsupported geometry type {}".format(str(type(geometry))))
        return None
    # print_time("Success move object {}".format(str(type(geometry))))
    return moved_geometry

def solid_union(breps):
    # 检查输入是否为空或包含少于两个Brep对象
    if not breps or len(breps) < 1:
        return None
    elif len(breps) == 1:
        return breps
    # 初始并集结果设为第一个Brep对象
    union_result = breps[0]

    # 逐个合并后续的Brep对象
    for brep in breps[1:]:
        # 执行布尔并集操作
        union_result = rg.Brep.CreateBooleanUnion([union_result, brep], 0.01)
        # 检查布尔操作是否成功
        if not union_result or len(union_result) == 0:
            return None
        union_result = union_result[0]

    return union_result

def translate_direction(normal):

    y = normal[1]
    if y != 0:
        xy = x / y
    else:
        xy = x * 999
    if normal[2] > 0.5:
        return 0
    elif y > 0 and xy >= -1 and xy < 1:
        return 1
    elif xy >= 1 or xy < -1:
        if x > 0:
            return 2
        if x < 0:
            return 4
    elif y < 0 and xy >= -1 and xy < 1:
        return 3

def brep_face_to_brep(brep_face):
    """
    Convert a BrepFace to a Brep.

    Args:
    brep_face (rg.BrepFace): The BrepFace to convert.

    Returns:
    rg.Brep: The new Brep created from the BrepFace.
    """
    # Duplicate the underlying Brep
    brep = brep_face.DuplicateFace(False)

    # Create a new Brep
    new_brep = rg.Brep()

    # Add the duplicated face to the new Brep
    new_brep.Append(brep)

    return new_brep

def points_near(point1, point2, tolerance=1e-6):
    if isinstance(point1, list):
        point1 = point1[0]
    try:
        point1 = rg.Point3d(point1)
        point2 = rg.Point3d(point2)

        distance = math.sqrt(
            (point1.X - point2.X) ** 2
            + (point1.Y - point2.Y) ** 2
            + (point1.Z - point2.Z) ** 2
        )
        return distance < tolerance
    except:
        raise KeyError(
            "Failed to get distance of points, Check input point1 {}, point2 {}".format(
                type(point1), type(point2)
            )
        )

def get_bottom(brep):
    faces = brep.Faces
    bottom = None
    lowest_z = float("inf")

    for face in faces:
        trimmed_surface = face.DuplicateFace(False)
        # 获取面的包围框
        bbox = face.GetBoundingBox(True)
        # 获取包围框的最低点
        min_point = bbox.Min

        # 找到 z 坐标最小的面
        if min_point.Z < lowest_z:
            lowest_z = min_point.Z
            bottom = trimmed_surface

    return bottom

def rotate(
    geometry,
    angle_degrees,
    rotation_plane=rg.Plane(rg.Point3d(0, 0, 0), rg.Vector3d(0, 0, 1)),
):
    # 将角度从度数转换为弧度
    angle_radians = math.radians(angle_degrees)

    # 从平面获取旋转中心和轴
    center = rotation_plane.Origin
    axis = rotation_plane.Normal

    # 创建旋转变换
    rotation_transform = rg.Transform.Rotation(angle_radians, axis, center)

    # 应用旋转变换到几何物体
    geometry.Transform(rotation_transform)

    return geometry
