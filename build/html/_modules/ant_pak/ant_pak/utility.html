def str_to_circle(str_geo, str_plane, str_radius):
    center_str = str_geo.replace("center ", "")
    center = str_to_point(center_str)
    flag1, value1 = str_to_flag(str_plane)
    value1 = int(value1)
    if flag1 == "plane":
        if value1 == 2:
            plane = rg.Plane.WorldXY
        elif value1 == 0:
            plane = rg.Plane.WorldYZ
        elif value1 == 1:
            plane = rg.Plane.WorldZX
    flag2, value2 = str_to_flag(str_radius)
    if flag2 == "radius":
        radius = value2
    circle = rg.Circle(plane, center, radius)
    return circle

def get_plane_from_points(points):
    """
    Check if a list of points are coplanar.
    """
    # Create a plane from the first three points
    plane = rg.Plane.FitPlaneToPoints([rg.Point3d(pt) for pt in points[:3]])[1]
    # Check if the remaining points lie on the plane
    for pt in points[3:]:
        if not plane.DistanceTo(pt) < 0.001:  # You can adjust the tolerance here
            return False
    return plane

def is_parallel_to_plane(points, plane_type):
    """
    Check if points are parallel to the specified plane.
    """
    # Define the normal vector of the specified plane type
    if plane_type == "xy":
        normal = rg.Vector3d.ZAxis
    elif plane_type == "yz":
        normal = rg.Vector3d.XAxis
    elif plane_type == "xz":
        normal = rg.Vector3d.YAxis
    else:
        return False

    # Calculate the plane that fits the points
    plane = get_plane_from_points(points)

    # Check if the plane normal is parallel to the specified plane type
    if plane:
        return plane.Normal.IsParallelTo(normal, RADIANS_TOL)
    else:
        return False

def get_corner_points(polyline):
    corner_points = [polyline.PointAt(0)]

    for i in range(polyline.SegmentCount - 1):
        current_segment = polyline.SegmentAt(i)
        next_segment = polyline.SegmentAt(i + 1)

        current_vector = current_segment.Direction
        next_vector = next_segment.Direction

        angle_radians = rg.Vector3d.VectorAngle(current_vector, next_vector)
        angle_degrees = math.degrees(angle_radians)

        if abs(angle_degrees) > 5:  # check if angle is not straight
            corner_points.append(polyline.PointAt(i + 1))

    return corner_points

def parallel_to_axis(plane):
    for axis in [plane.XAxis, plane.YAxis]:
        all_axis = [rg.Vector3d.XAxis, rg.Vector3d.YAxis, rg.Vector3d.ZAxis]
        bools = [axis.IsParallelTo(to_axis, RADIANS_TOL) for to_axis in all_axis]

        if sum([abs(i) for i in bools]) == 0:
            continue
        elif sum([abs(i) for i in bools]) == 1:
            index = [index for index, value in enumerate(bools) if abs(value) == 1][0]
            to_axis = all_axis[index]
            print("plane is aligned with {}".format(str(to_axis)))
            return to_axis
        elif sum([abs(i) for i in bools]) == 2:
            print("plane is not rotated")
            return False

    print("plane is rotated around an arbitrary axis")
    return None

def get_rectangle_rotated_axis(rectangle):
    """
    Check if a rectangle is rotated around x, y, or z axis.
    Return rotation axis, direction and angle if rotated, otherwise return None.
    """
    # Check if rectangle is oriented aligned with world axes
    plane = rectangle.Plane
    rotation_axis = parallel_to_axis(plane)
    if rotation_axis:
        #        rotation_direction = rectangle.Plane.ZAxis.IsParallelTo(rotation_axis)
        if rotation_axis == rg.Vector3d.ZAxis:
            angle = math.acos(rectangle.Plane.ZAxis * rg.Plane.WorldXY.XAxis)
            index_axis = 2
        elif rotation_axis == rg.Vector3d.YAxis:
            angle = math.acos(rectangle.Plane.ZAxis * rg.Plane.WorldXY.ZAxis)
            index_axis = 1
        elif rotation_axis == rg.Vector3d.XAxis:
            angle = math.acos(rectangle.Plane.ZAxis * rg.Plane.WorldXY.YAxis)
            index_axis = 0

        if angle >= 0:
            rotation_direction = 1
        else:
            rotation_direction = -1

        return index_axis, rotation_direction, math.degrees(angle) + 90
    else:
        return False

def polyline_to_rectangle(polyline):

    tol = 0.01
    # lines = polyline.GetSegments()
    points = get_polyline_points(polyline)

    print("1- Polyline points is {}".format(len(points)))
    # Step 1: Check if the curve has 4 lines
    if len(points) != 4:
        return polyline

    # Step 2: Check if opposite sides are parallel and of equal length
    side_lengths = []
    side_vectors = []

    for i in range(4):
        start_pt = points[i]
        end_pt = points[i + 1 if i < 3 else 0]
        # if i==3:
        #     end_pt = points[0]
        # else:
        #     end_pt = points[i+1]
        side_vector = end_pt - start_pt
        side_length = side_vector.Length
        side_lengths.append(side_length)
        side_vectors.append(side_vector)

    #  Check if opposite sides are parallel

    if (
        side_vectors[0].IsParallelTo(side_vectors[2], RADIANS_TOL) != -1
        or side_vectors[1].IsParallelTo(side_vectors[3], RADIANS_TOL) != -1
    ):
        print("2-Opposite sides are not parallel")
        return polyline

    # Check if opposite sides are of equal length
    if (
        abs(side_lengths[0] - side_lengths[2]) > tol
        or abs(side_lengths[1] - side_lengths[3]) > tol
    ):
        print("4-Opposite sides are not of equal length")
        return polyline

    # Step 3: Check if Adjacent sides are Perpendicular
    if side_vectors[1].IsPerpendicularTo(side_vectors[0], RADIANS_TOL) is False:
        print("3- Adjacent sides are not Perpendicular")
        return polyline

    print("The given polyline is a rectangle.")
    # points = [line.From for line in lines]

    # Create a rectangle
    base_plane = rg.Plane.FitPlaneToPoints(points[:3])[1]
    rectangle = rg.Rectangle3d(base_plane, points[0], points[2])
    return rectangle

def vector_parallel_to_axis(vector):
    # 检查向量是否与X轴、Y轴或Z轴平行
    if is_vector_parallel_to_axis(vector, rg.Vector3d.XAxis):
        return 1 if vector.X > 0 else -1
    elif is_vector_parallel_to_axis(vector, rg.Vector3d.YAxis):
        return -1 if vector.Y > 0 else 1
    elif is_vector_parallel_to_axis(vector, rg.Vector3d.ZAxis):
        return 1 if vector.Z > 0 else -1
    else:
        return False

def plane_distance(plane1, plane2):
    # 提取两个平面的原点
    # origin1 = plane1.Origin
    # origin2 = plane2.Origin

    # 提取平面的法向量
    normal1 = plane1.Normal
    normal2 = plane2.Normal

    # 检查平面是否平行（法向量是否平行）
    if not normal1.IsParallelTo(normal2):
        raise ValueError("The planes are not parallel.")

    # 判断是否平行于X\Y\Z轴
    if not vector_parallel_to_axis(normal1):
        raise ValueError("The planes are not parallel to any axis")

    # 获取第一个平面的原点
    point_on_plane1 = plane1.Origin

    # 计算第一个平面上的点到第二个平面的距离
    distance = abs(plane2.DistanceTo(point_on_plane1))

    # 返回距离的绝对值
    return distance

def get_brep_bottom_and_height(brep):
    """
    Optimized function to find the bottom edge curve and height of a Brep object.

    Parameters:
        - brep: The Brep object to analyze.

    Returns:
        - A tuple containing the outer loop curve of the bottom face and its height relative to another parallel face,
      or None if no parallel faces are found.
    """
    faces = brep.Faces
    faces = sorted(faces, key=lambda face: get_area(face), reverse=True)
    for i in range(brep.Faces.Count):
        face1 = faces[i]
        plane1 = face1.TryGetPlane()[1]
        area1 = rg.AreaMassProperties.Compute(face1).Area
        if not plane1:
            continue

        for j in range(i + 1, brep.Faces.Count):
            face2 = faces[j]
            plane2 = face2.TryGetPlane()[1]
            area2 = rg.AreaMassProperties.Compute(face2).Area
            if not plane2:
                continue
            if (
                are_vectors_parallel_dot(plane1.Normal, plane2.Normal)
                and abs(area1 - area2) < 0.01
            ):
                # print area1, area2
                distance = plane_distance(plane1, plane2)

                print_time(
                    "vectors({} and {}) are parallel, plane distance is {}".format(
                        plane1.Normal, plane2.Normal, distance
                    )
                )
                if distance:
                    if (
                        plane1.Origin.X + plane1.Origin.Y + plane1.Origin.Z
                        < plane2.Origin.X + plane2.Origin.Y + plane2.Origin.Z
                    ):
                        base_face = face1
                    else:
                        base_face = face2
                    height = round(distance, 2)
                    # Assuming the first face found is the bottom one
                    return base_face.OuterLoop.To3dCurve(), height

    return None, 0

def are_vectors_parallel_dot(v1, v2):
    # 归一化向量
    v1.Unitize()
    v2.Unitize()

    # 计算点积
    dot_product = rg.Vector3d.Multiply(v1, v2)
    # print "dot_product",dot_product,"XXXXXXXXXXXXXXXXXXX"

    # 自定义比较函数
    def isclose(a, b, rel_tol=1e-9, abs_tol=0.0):
        # print abs(a-b), max(rel_tol * max(abs(a), abs(b)), abs_tol)
        return abs(a - b) <= max(rel_tol * max(abs(a), abs(b)), abs_tol)

    # 判断点积是否接近1或-1
    return isclose(dot_product, 1.0, abs_tol=1e-2) or isclose(
        dot_product, -1.0, abs_tol=1e-2
    )

def find_paraview_exe_file():
    """
    查找指定目录及其子目录下的paraview..exe文件
    """
    dir_paths = get_common_program_files()
    for dir_path in dir_paths:
        for filename in os.listdir(dir_path):
            # print dir_path,filename
            if "ParaView" in filename:
                return dir_path + "\\" + filename + "\\bin\\paraview.exe"
    raise ValueError("No exe file found in the folder.")
