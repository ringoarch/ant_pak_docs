def load_project(project):
    """
    加载项目。

    :param project: 项目路径
    :return object_strings: 对象字符串列表
    """
    files = os.listdir(project)
    object_strings = []
    for file in files:
        if file == "model":
            file_path = os.path.join(project, file)
            with open(file_path, "r") as f:
                lines = f.readlines()
                object_string = ""
                start = False
                for line in lines:
                    # 在这里处理每一行的内容，可以打印或者进行其他操作
                    if "object" in line and "end" not in line:
                        object_string += line
                        start = True
                    elif "end object" in line:
                        object_strings.append(object_string)
                        object_string = ""
                        start = False
                    elif start:
                        object_string += line
    return object_strings

def read_object_string(object_string):
    """
    读取对象字符串，返回对象。

    :param object_string: 对象字符串
    :return object: 对象
    """
    strings = object_string.split("\n")
    header = strings[0]
    body_settings = []
    for string in strings:
        if "shape" in string and "end" not in string:
            shape_header = string
        elif "setval" in string:
            shape_body = string
        elif "object" not in string:
            body_settings.append(string)
    [object, object_type, object_name] = header.split(" ")
    mesh_type = shape_header.split(" ")[-1]
    flags = shape_body.replace("    setval ", "").split("} ")[:-1]
    flags = [i + "}" for i in flags]
    dict_flags = flags_to_dict(flags)
    if "diff" in shape_body:
        [x, y, z] = dict_flags["diff"]
        if x == 0:
            plane = rg.Plane.WorldXY
            geo_type = "rectangle"
        elif y == 0:
            plane = rg.Plane.WorldYZ
            geo_type = "rectangle"
        elif z == 0:
            plane = rg.Plane.WorldZX
            geo_type = "rectangle"
        else:
            geo_type = "box"
    # elif
    if object_type == "opening" and geo_type == "rectangle":
        object = OpeningRectangle()

    return object

def flags_to_dict(flags):
    """
    标志转换为字典。

    :param flags: 标志列表
    :return dict: 字典
    """
    dict = {}
    for flag in flags:
        [key, value] = flag.split(" ")
        dict[key] = value.replace("{", "").replace("}", "")
    return dict
