class Partition(_BaseObj, TemperatureMixIn, ThermalMixIn):
    """
    Properties:
        * identifier
        * display_name
    """

    # __slots__ = ("display_name",)

    def __init__(self):
        """
        初始化Partition对象。

        该方法调用父类的初始化方法，并设置一些默认属性。
        """
        _BaseObj.__init__(self)  # process the identifier
        TemperatureMixIn.__init__(self)

        self.display_name = "partition.{}".format(self.identifier)
        self.obj_type = "plate"
        self.species_enabled = 0
        self.free_temperature = AMBIENT_TEMP
        self.current_genus = "free"

        self.vtype = "stationary"
        self.eff_thick = 0
        self.thermal_rtype = "reftemp"
        self.eff_thick_on = 0
        self.thermal_itemp = AMBIENT_TEMP
        self.cres_heat_tr_on = 1
        self.plate_type = "adiabatic_thin"

    @property
    def vtype(self):
        """
        隔断的vtype。

        * stationery:静态
        * moving:动态

        :return: vtype值
        """
        return self.vtype

    @vtype.setter
    def vtype(self, value):
        """
        :param value: vtype值
        :return: None
        """
        if value in VTYPE:
            self.vtype = value
        else:
            raise ValueError("Vtype must in {}".format(str(VTYPE)))

    def set_thick(self, partition_thickness=20, plate_type="adiabatic_thick"):
        """
        设置分区厚度

        :param partition_thickness: 分区厚度

        :param plate_type: 分区类型,默认为adiabatic_thick

        :return: None
        """

        self.eff_thick = partition_thickness
        # self.eff_thick_on=eff_thick_on
        self.plate_type = plate_type
        self.add_body_settings(["eff_thick", "plate_type"])
        return self





[文档]
    def set_itemp(self, itemp=26, unit="C"):
        """
        设置分区内温度

        :param itemp: 分区内温度
        :param unit: 单位,默认为摄氏度

        :return: 当前对象
        """

        self.add_body_settings(["thermal_itemp", "thermal_type", "thermal_rtype"])
        self.thermal_type = "temp"
        self.thermal_itemp = ABS_TEMP + itemp
        return self

    def set_out_flux(
        self,
        value,
    ):
        """
        设置分区外热流

        :param value: 热流
        :return: 当前对象
        """
        self.thermal_heat_area = value
        self.add_body_settings(["thermal_heat_area"])
        self.remove_body_settings(["thermal_type"])
        return self

class PartitionRectangle(Partition, RectangleMixIn):

    def __init__(self):
        Partition.__init__(self)
        RectangleMixIn.__init__(self)

    @classmethod
    def from_rectangle(self, rectangle):
        """
        :param rectangle: 输入的Rectangle对象。
        :return: 创建的PartitionRectangle对象。
        """
        partition = PartitionRectangle()
        partition = partition.from_rect(rectangle)
        return partition
