class Block(_BaseObj, ThermalMixIn, SpeciesMixIn, RadiationMixIn):
    """
    Properties:
        * identifier
        * display_name
    """

    # __slots__ = ("display_name",)

    def __init__(self):
        """
        初始化Block对象。

        该方法调用父类的初始化方法，并设置一些默认属性。
        """
        _BaseObj.__init__(self)  # process the identifier
        ThermalMixIn.__init__(self)
        SpeciesMixIn.__init__(self)
        RadiationMixIn.__init__(self)
        self.obj_type = "block"
        self.current_stype = STYPE[0]
        self.current_genus = "default"
        self.display_name = "block.{}".format(self.identifier)
        self.block_type = "hollow"
        self.add_body_settings(["current_genus", "block_type"])



[文档]
    def set_all_temp(self, all_temp=26, unit="C"):
        """
        设置所有温度相关的属性。

        :param all_temp: 温度值，默认为26。
        :param unit: 温度单位，默认为"C"（摄氏度）。

        :return self: 返回当前对象，以便进行链式调用。
        """
        self.add_body_settings(
            ["all_temp", "thermal_type", "all_thermal_type", "thermal_rtype"]
        )
        self.thermal_type = "temp"
        self.all_temp = ABS_TEMP + all_temp
        self.all_thermal_type = "temp"
        self.thermal_rtype = "reftemp"
        return self
