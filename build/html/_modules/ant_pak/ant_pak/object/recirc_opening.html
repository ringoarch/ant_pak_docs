class RecircOpening(_BaseObj, MassFlowMixIn, HeatFlowMixIn):

    def __init__(self):
        """
        初始化RecircOpening对象。

        该方法调用父类的初始化方法，并设置一些默认属性。
        """
        _BaseObj.__init__(self)  # process the identifier
        MassFlowMixIn.__init__(self)
        HeatFlowMixIn.__init__(self)

        self.display_name = "recirc opening.{}".format(self.identifier)
        self.current_genus = "recirc"
        self.obj_type = "opening"
        self.supply = None
        self.extract = None
        self.current_section = "supply"
        self.body_settings = ["creation_order", "current_genus", "current_section"]

    def from_crvs(self, supply, extract):
        """
        从送风和回风口创建一个RecircOpening对象。

        :param supply: 送风口。
        :param extract: 回风口。

        :return RecircOpening: 创建的RecircOpening对象。
        """
        self.supply = create_opening(supply)
        self.extract = create_opening(extract)
        self.supply.shape_type = "supply"
        self.extract.shape_type = "extract"
        self.geometry = [self.supply.geometry, self.extract.geometry]
        self.point = self.supply.point
        return self

    def to_str(self, creation_order=0, param_on=False):
        """
        将RecircOpening对象转换为字符串表示。

        :param creation_order: 对象的创建顺序。
        :param param_on: 是否启用参数。

        :return str: RecircOpening对象的字符串表示。
        """
        begin = self.str_begin()
        self.creation_order = creation_order
        str_supply = self.clear_string(self.supply.to_str()).replace(
            "body_shape", "supply_shape"
        )
        str_extract = self.clear_string(self.extract.to_str()).replace(
            "body_shape", "extract_shape"
        )
        str_body_settings = self.str_by_setting()
        body = str_supply + str_extract + str_body_settings
        end = self.str_end()
        string = begin + body + end
        return string

    def clear_string(self, string):
        """
        清除RecircOpening对象的字符串表示。

        :param string: 要清除的字符串。

        :return str: 清除后的字符串。
        """
        split_strings = string.split("\n")
        new_strings = []
        for str in split_strings:
            if (
                "object opening" in str
                or "end object" in str
                or "creation_order" in str
                or "current_genus" in str
            ):
                continue
            else:
                new_strings.append(str)

        new_string = ""
        for n_str in new_strings:
            new_string += n_str + "\n"
        return new_string
