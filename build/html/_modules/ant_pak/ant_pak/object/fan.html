class Fan(_BaseObj, FlowMixIn, SwirlMixIn, SpeciesMixIn):
    """
    Properties:
        * identifier
        * display_name
    """

    # __slots__ = ("display_name",)

    def __init__(self):
        """
        初始化Fan对象。

        该方法调用父类的初始化方法，并设置一些默认属性。
        """
        _BaseObj.__init__(self)  # process the identifier
        FlowMixIn.__init__(self)
        SwirlMixIn.__init__(self)
        SpeciesMixIn.__init__(self)
        self.display_name = "fan.{}".format(self.identifier)
        self.obj_type = "fan"
        self.current_genus = "default"
        self.dir_spec = "inward"
        self.add_body_settings(["current_genus", "dir_spec"])
