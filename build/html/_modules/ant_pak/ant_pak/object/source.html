class Source(_BaseObj, TemperatureMixIn, SpeciesMixIn):
    """
    Properties:
        * identifier
        * display_name
        * base_point
        * rooms
        * geometry
        * story
        * height
        * repeat
    """

    # __slots__ = (
    #     "display_name",
    #     "base_point",
    #     "rooms",
    #     "area",
    #     "parent",
    #     "level",
    #     "geometry",
    #     "story",
    #     "height",
    #     "repeat",
    #     "apertures",
    #     "outline",
    #     "net_geometry",
    # )

    def __init__(self):
        """
        初始化Source对象。

        该方法调用父类的初始化方法，并设置一些默认属性。
        """
        _BaseObj.__init__(self)  # process the identifier
        TemperatureMixIn.__init__(self)
        SpeciesMixIn.__init__(self)

        self.obj_type = "source"

        self.current_stype = STYPE[0]
        self.shape_begin = "body_shape shape_" + self.current_stype

        self.current_genus = "default"
        self.display_name = "source.{}".format(self.identifier)

        self.add_body_settings(["current_stype", "current_genus"])
