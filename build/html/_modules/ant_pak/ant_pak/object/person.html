    def from_point(self, point, posture="standing", height=1.73, facing_dir=1):
        self.point = point
        self.posture = posture
        self.height = height
        if posture == "sitting":
            self.lap_fraction = 0.15
        else:
            self.lap_fraction = 0

        self.bot_x = point.X
        self.bot_y = point.Y
        # 头部
        head_point1, head_point2 = self.points_from_point(
            point,
            0.08,
            0.08,
            self.height,
            1 - self.head_fraction,
            1,
            facing_dir=facing_dir,
        )
        self.head = BlockPrism.from_points(head_point1, head_point2)
        self.head.shape_type = "head"
        self.head.shape_begin = (
            self.head.shape_type + "_shape shape_" + self.head.current_stype
        )
        # 身体
        if facing_dir == 0:
            body_point1, body_point2 = self.points_from_point(
                point,
                self.body_depth / 2,
                self.body_width / 2,
                self.height,
                1 - (self.head_fraction + self.body_fraction),
                1 - self.head_fraction,
                facing_dir=facing_dir,
            )
        else:
            body_point1, body_point2 = self.points_from_point(
                point,
                self.body_width / 2,
                self.body_depth / 2,
                self.height,
                1 - (self.head_fraction + self.body_fraction),
                1 - self.head_fraction,
                facing_dir=facing_dir,
            )
        self.body = PrismMixIn().from_points(body_point1, body_point2)
        self.body.shape_type = "body"
        self.body.shape_begin = (
            self.body.shape_type + "_shape shape_" + self.body.current_stype
        )
        # 左右手
        if facing_dir == 0:
            larm_point = rg.Point3d(
                point.X, point.Y + self.body_width / 2 + 0.05, point.Z
            )
        else:
            larm_point = rg.Point3d(
                point.X + self.body_width / 2 + 0.05, point.Y, point.Z
            )
        larm_point1, larm_point2 = self.points_from_point(
            larm_point,
            0.05,
            0.05,
            self.height,
            0.4778,
            1 - self.head_fraction,
            facing_dir=facing_dir,
        )
        self.larm = PrismMixIn().from_points(larm_point1, larm_point2)
        self.larm.shape_type = "larm"
        self.larm.shape_begin = (
            self.larm.shape_type + "_shape shape_" + self.larm.current_stype
        )
        if facing_dir == 0:
            rarm_point = rg.Point3d(
                point.X, point.Y - self.body_width / 2 - 0.05, point.Z
            )
        else:
            rarm_point = rg.Point3d(
                point.X - self.body_width / 2 - 0.05, point.Y, point.Z
            )
        rarm_point1, rarm_point2 = self.points_from_point(
            rarm_point,
            0.05,
            0.05,
            self.height,
            0.4778,
            1 - self.head_fraction,
            facing_dir=facing_dir,
        )
        self.rarm = PrismMixIn().from_points(rarm_point1, rarm_point2)
        self.rarm.shape_type = "rarm"
        self.rarm.shape_begin = (
            self.rarm.shape_type + "_shape shape_" + self.rarm.current_stype
        )
        # 左右脚
        if facing_dir == 0:
            lleg_point = rg.Point3d(
                point.X, point.Y + self.body_width / 2 - 0.05, point.Z
            )
        else:
            lleg_point = rg.Point3d(
                point.X + self.body_width / 2 - 0.05, point.Y, point.Z
            )
        lleg_point1, lleg_point2 = self.points_from_point(
            lleg_point,
            0.05,
            0.05,
            self.height,
            0.0,
            1 - (self.head_fraction + self.body_fraction),
            "leg",
            facing_dir=facing_dir,
        )
        self.lleg = PrismMixIn().from_points(lleg_point1, lleg_point2)
        self.lleg.shape_type = "lleg"
        self.lleg.shape_begin = (
            self.lleg.shape_type + "_shape shape_" + self.lleg.current_stype
        )
        if facing_dir == 0:
            rleg_point = rg.Point3d(
                point.X, point.Y - self.body_width / 2 + 0.05, point.Z
            )
        else:
            rleg_point = rg.Point3d(
                point.X - self.body_width / 2 + 0.05, point.Y, point.Z
            )
        rleg_point1, rleg_point2 = self.points_from_point(
            rleg_point,
            0.05,
            0.05,
            self.height,
            0.0,
            1 - (self.head_fraction + self.body_fraction),
            "leg",
            facing_dir=facing_dir,
        )
        self.rleg = PrismMixIn().from_points(rleg_point1, rleg_point2)
        self.rleg.shape_type = "rleg"
        self.rleg.shape_begin = (
            self.rleg.shape_type + "_shape shape_" + self.rleg.current_stype
        )
        # 脸
        face_point = rg.Point3d(point.X, point.Y + 0.08, point.Z)
        face_point1, face_point2 = self.points_from_point(
            face_point, 0.08, 0, self.height, 0.865, 1, facing_dir=facing_dir
        )

        self.face = RectangleMixIn().from_plane_and_points(
            rg.Plane.WorldZX, face_point1, face_point2
        )
        self.face.shape_type = "face"
        self.face.shape_begin = (
            self.face.shape_type + "_shape shape_" + self.face.current_stype
        )

        if posture == "sitting":
            self.add_body_settings(["posture"])
            lthigh_point1, lthigh_point2 = self.points_from_point(
                lleg_point,
                0.05,
                0.05,
                self.height,
                1 - (self.head_fraction + self.body_fraction),
                1 - (self.head_fraction + self.body_fraction) + 0.1 / self.height,
                "thigh",
                facing_dir=facing_dir,
            )
            self.lthigh = BlockPrism.from_points(lthigh_point1, lthigh_point2)
            self.lthigh.shape_type = "lthigh"
            self.lthigh.shape_begin = (
                self.lthigh.shape_type + "_shape shape_" + self.lthigh.current_stype
            )

            rthigh_point1, rthigh_point2 = self.points_from_point(
                rleg_point,
                0.05,
                0.05,
                self.height,
                1 - (self.head_fraction + self.body_fraction),
                1 - (self.head_fraction + self.body_fraction) + 0.1 / self.height,
                "thigh",
                facing_dir=facing_dir,
            )
            self.rthigh = BlockPrism.from_points(rthigh_point1, rthigh_point2)
            self.rthigh.shape_type = "rthigh"
            self.rthigh.shape_begin = (
                self.rthigh.shape_type + "_shape shape_" + self.rthigh.current_stype
            )
        else:
            self.lap_fraction = 0
            self.remove_body_settings(["posture"])

        self.geometry = [
            self.head.geometry,
            # self.face.geometry,
            self.body.geometry,
            self.larm.geometry,
            self.rarm.geometry,
            self.lleg.geometry,
            self.rleg.geometry,
        ]
        if posture == "sitting":
            self.geometry += [self.lthigh.geometry, self.rthigh.geometry]
        self.geo_to_str()
        return self

    def set_met(self, met_type, value):
        self.met_type = met_type
        if met_type == "total":
            self.met_total = value
            self.add_body_settings(["met_type", "met_total"])
            self.add_body_settings(["met_temp"])
        elif met_type == "":
            self.met_rate = value
            self.remove_body_settings(["met_type", "met_total"])
        elif met_type == "temp":
            self.add_body_settings(["met_type", "met_temp"])
            self.remove_body_settings(["met_total"])
            self.met_temp = value + ABS_TEMP
        return self

    def geo_to_str(self):
        head_shape = self.head.geo_to_str()
        face_shape = self.face.geo_to_str()
        body_shape = self.body.geo_to_str()
        larm_shape = self.larm.geo_to_str()
        rarm_shape = self.rarm.geo_to_str()
        lleg_shape = self.lleg.geo_to_str()
        rleg_shape = self.rleg.geo_to_str()
        shape = (
            head_shape
            + face_shape
            + body_shape
            + larm_shape
            + rarm_shape
            + lleg_shape
            + rleg_shape
        )
        if self.posture == "sitting":
            lthigh_shape = self.lthigh.geo_to_str()
            rthigh_shape = self.rthigh.geo_to_str()
            shape += lthigh_shape + rthigh_shape
        self.shape = shape
        return shape

    def points_from_point(
        self,
        point,
        offset_w,
        offset_d,
        height,
        fraction_1,
        fraction_2,
        type="",
        facing_dir=1,
    ):
        base_x, base_y, base_z = point.X, point.Y, point.Z
        if facing_dir == 1:
            if self.posture == "sitting" and type == "leg":
                dy_1 = 0.3095
                dy_2 = 0.3095
                dx_1 = 0
                dx_2 = 0
            elif self.posture == "sitting" and type == "thigh":
                dy_1 = 0.15
                dy_2 = 0.3095
                dx_1 = 0
                dx_2 = 0
            else:
                dy_1 = 0
                dy_2 = 0
                dx_1 = 0
                dx_2 = 0
        else:
            if self.posture == "sitting" and type == "leg":
                dx_1 = 0.3095
                dx_2 = 0.3095
                dy_1 = 0
                dy_2 = 0
            elif self.posture == "sitting" and type == "thigh":
                dx_1 = 0.15
                dx_2 = 0.3095
                dy_1 = 0
                dy_2 = 0
            else:
                dx_1 = 0
                dx_2 = 0
                dy_1 = 0
                dy_2 = 0
        point1 = rg.Point3d(
            base_x - offset_w + dx_1,
            base_y - offset_d + dy_1,
            base_z + height * (fraction_1 - self.lap_fraction),
        )
        point2 = rg.Point3d(
            base_x + offset_w + dx_2,
            base_y + offset_d + dy_2,
            base_z + height * max((fraction_2 - self.lap_fraction), 0),
        )
        return point1, point2
