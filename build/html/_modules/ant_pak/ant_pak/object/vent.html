class Vent(_BaseObj, VelocityMixIn, TemperatureMixIn):
    """
    Properties:
        * identifier
        * display_name
        * base_point
        * rooms
        * geometry
        * story
        * height
        * repeat
    """

    # __slots__ = ("display_name",)

    def __init__(self):
        """
        初始化Vent对象。

        该方法调用父类的初始化方法，并设置一些默认属性。
        """
        _BaseObj.__init__(self)  # process the identifier
        VelocityMixIn.__init__(self)
        TemperatureMixIn.__init__(self)
        self.display_name = "vent.{}".format(self.identifier)
        self.obj_type = "ventres"
        self.species_enabled = 0
        self.free_temperature = AMBIENT_TEMP
        self.current_stype = STYPE[1]
        self.shape_begin = "body_shape shape_" + self.current_stype
        self.current_genus = "free"
        self.vent_type = "internal"
        self.add_body_settings(["vent_type"])
        self.xratioq = 1
