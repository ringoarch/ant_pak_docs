    def set_species(self, names):
        """
        设置污染物名称。

        :param names: 要设置的污染物名称列表。
        """
        value = len(names) + 1
        self.enabled = 1
        self.count = value
        self.actives = [1] * value
        self.names = ["Air"] + names
        initvals = [0] + [DEFAULT_SPECIES[name] for name in names]
        self.initvals = initvals
        self.types = ["fraction"] * value
        self.actives = [1] * value

    def to_str(self):
        """
        返回污染物对象的字符串表示。

        :return: 污染物对象的字符串表示。
        """
        string = ""
        string += "    enabled {}\n".format(self.enabled)
        string += "    count {}\n".format(self.count)
        string += "    maxnum {}\n".format(self.maxnum)
        string += "    available {}\n".format(self.available)
        for index, name in enumerate(self.names):
            num = index + 1
            add_name = "    name{} {}\n".format(str(num), name)
            add_type = "    type{} {}\n".format(str(num), self.types[index])
            add_initval = "    initval{} {}\n".format(str(num), self.initvals[index])
            add_active = "    active{} {}\n".format(str(num), self.actives[index])

            string += add_name
            string += add_type
            string += add_initval
            string += add_active
        self.shape = string
        return string
