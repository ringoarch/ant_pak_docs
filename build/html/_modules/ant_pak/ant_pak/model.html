def copy_problem(work_folder):
    """
    复制问题文件

    :param work_folder: 工作文件夹
    :return problem: 问题文件
    """
    dir = os.path.dirname(os.path.abspath(__file__))
    problem = os.path.join(dir, "resources\problem")
    current_problem = os.path.join(work_folder, "problem")
    if os.path.exists(current_problem):
        pass
    else:
        shutil.copy(problem, work_folder)
    return os.path.join(work_folder, "problem")

def create_opening_from_crv(crv):
    """
    从曲线创建开口，支持矩形、多段线、圆形。

    * 如果是矩形，返回OpeningRectangle；
    * 如果是多段线，返回OpeningPolygon；
    * 如果是圆形，返回OpeningCircle；
    * 如果是其他类型，返回None。

    :param crv: 曲线
    :return opening: 开口或者None
    """
    if isinstance(crv, rg.Circle):
        return create_opening_from_circle(crv)
    elif (
        isinstance(crv, rg.PolyCurve)
        or isinstance(crv, rg.Polyline)
        or isinstance(crv, rg.PolylineCurve)
        or isinstance(crv, rg.NurbsCurve)
    ):
        polyline = convert_to_polyline(crv)
        rectangle = polyline_to_rectangle(polyline)
        if isinstance(rectangle, rg.Rectangle3d):
            return create_opening_from_rectangle(rectangle)
        else:
            return create_opening_from_polyline(rectangle)
    elif isinstance(crv, rg.Rectangle3d):
        return create_opening_from_rectangle(crv)
    return None

def create_opening_from_rectangle(rectangle):
    """
    从矩形创建开口，支持水平矩形、垂直矩形。

    * 如果是水平矩形，返回OpeningRectangle；
    * 如果是垂直矩形，返回OpeningIncline；
    * 如果是其他类型，返回None。

    :param rectangle: 矩形
    :return opening: 开口
    """
    normal = rectangle.Plane.Normal
    if vector_parallel_to_axis(normal):
        opening = OpeningRectangle.from_rectangle(rectangle)
    else:
        opening = OpeningIncline.from_rectangle(rectangle)
    return opening

def create_source_from_person(person):
    head = person.head
    n = 0.05
    point1 = rg.Point3d(head.point1.X - n, head.point1.Y - n, head.point1.Z - n)
    point2 = rg.Point3d(head.point2.X + n, head.point2.Y + n, head.point2.Z + n)
    box = rg.BoundingBox([point1, point2])
    source = SourcePrism.from_box(box)
    source.visiable = 0
    return source

def object_instance(obj):
    """
    获取对象的实例类型，支持Domain、Wall、Partition、Opening、Vent、Block、Source、Fan、Person、Resistance。

    :param obj: 对象
    :return: 实例类型
    """
    if isinstance(obj, Domain):
        return "domain"
    elif isinstance(obj, Wall):
        return "wall"
    elif isinstance(obj, Partition):
        return "partition"
    elif isinstance(obj, Opening):
        return "opening"
    elif isinstance(obj, Vent):
        return "vent"
    elif isinstance(obj, Block):
        return "block"
    elif isinstance(obj, Source):
        return "source"
    elif isinstance(obj, Fan):
        return "fan"
    elif isinstance(obj, Person):
        return "person"
    elif isinstance(obj, Resistance):
        return "resistance"
    else:
        raise ValueError("Wrong object")

def get_brep_parallel(brep):
    """
    获取BREP的平行面。

    :param brep: BREP
    :return: x_parallel, y_parallel, z_parallel
    """
    tol = 0.001
    x_parallel = False
    y_parallel = False
    z_parallel = False

    normals = [face.Normal for face in brep.Faces]
    if len([normal.X for normal in normals if abs(normal.X) > 1 - tol]) == 2:
        x_parallel = True
    elif len([normal.Y for normal in normals if abs(normal.Y) > 1 - tol]) == 2:
        y_parallel = True
    elif len([normal.Z for normal in normals if abs(normal.Z) > 1 - tol]) == 2:
        z_parallel = True

    return x_parallel, y_parallel, z_parallel
