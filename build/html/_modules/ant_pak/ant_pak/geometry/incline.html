class InclineMixIn(_BaseGeo):
    """
    倾斜面混合类。
    该类继承自_BaseGeo类，并添加了一些特定于倾斜面的属性和方法。

        _BaseGeo.__init__(self)

        self.point1 = (0, 0, 0)
        self.point2 = (5, 5, 3)
        self.current_stype = "incline"
        self.shape_begin = self.shape_type + "_shape shape_" + self.current_stype
        self.axis = 1
        self.rotate_sign = -1
        self.rotate_angle = 0

    def from_rect(self, rectangle):
        """
        从矩形对象创建倾斜面。

        该方法接收一个矩形对象作为参数，并将其转换为倾斜面的几何信息。倾斜面的几何信息包括中心点、半径、高度等。

        :param rectangle: 矩形对象。
        :return InclineMixIn: 倾斜面对象。
        """
        self.geometry = rectangle
        self.refresh_point()
        point1 = rectangle.Corner(0)
        point2 = rectangle.Corner(2)
        self.point1 = point1
        self.point2 = point2

        result = get_rectangle_rotated_axis(rectangle)

        if result:
            rotation_axis, rotation_direction, angle = result
            print_time(
                "{} Type {}\nRotation axis: {}\nRotation direction: {}\nRotation angle: {} degrees".format(
                    self.display_name,
                    type(rectangle),
                    rotation_axis,
                    "Clockwise" if rotation_direction < 0 else "Counter-clockwise",
                    angle,
                )
            )
            self.axis = rotation_axis
            self.rotate_sign = rotation_direction
            self.rotate_angle = angle
            self.geo_to_str()
            return self
        else:
            return None

    def from_incline(self, polyline):
        """
        从倾斜面对象创建倾斜面。

        该方法接收一个倾斜面对象作为参数，并将其转换为倾斜面的几何信息。倾斜面的几何信息包括中心点、半径、高度等。

        :param incline: 倾斜面对象。
        :return InclineMixIn: 倾斜面对象。
        """
        self.geometry = polyline
        self.refresh_point()
        result = get_polyline_rotated_axis(polyline)
        if result:
            rotation_axis, rotation_direction, angle = result
            print_time(
                "{} Type {}\nRotation axis: {}\nRotation direction: {}\nRotation angle: {} degrees".format(
                    self.display_name,
                    type(polyline),
                    rotation_axis,
                    "Clockwise" if rotation_direction < 0 else "Counter-clockwise",
                    angle,
                )
            )
            self.axis = rotation_axis
            self.rotate_sign = rotation_direction
            self.rotate_angle = angle
            self.verts = get_polyline_points(polyline)
            self.height = 0
            self.nverts = len(self.verts)
            self.geo_to_str()

    def geo_to_str(self):
        """
        将倾斜面的几何信息转换为字符串表示形式。

        该方法将倾斜面的中心点、半径、高度等信息转换为字符串，并将这些字符串拼接成一个完整的形状字符串。

        :return str: 表示倾斜面几何信息的字符串。
        """
        str_geo = points_to_str(self.point1, self.point2)

        str_volume_flag = flag_to_str("volume_flag", self.volume_flag)
        str_diff_flag = flag_to_str("diff_flag", self.diff_flag)
        str_split_flag = flag_to_str("split_flag", self.split_flag)

        str_axis = flag_to_str("axis", self.axis)
        str_rotate_sign = flag_to_str("rotate_sign", self.rotate_sign)
        str_rotate_angle = flag_to_str("rotate_angle", self.rotate_angle)
        str_thickness = flag_to_str("thickness", self.thickness)
        flags = [
            str_volume_flag,
            str_diff_flag,
            str_split_flag,
            str_axis,
            str_rotate_sign,
            str_rotate_angle,
            str_thickness,
        ]

        str_flags = list_str(flags)

        shape_begin = "    shape {}\n".format(self.shape_begin)
        shape_body = "        setval {} {}\n".format(str_geo, str_flags)
        shape_end = "    end shape\n"
        shape = shape_begin + shape_body + shape_end
        self.shape = shape
        return shape
