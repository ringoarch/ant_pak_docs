    def geo_to_str(self):
        """
        将多边形的几何信息转换为字符串表示形式。

        该方法将多边形的中心点、半径、高度等信息转换为字符串，并将这些字符串拼接成一个完整的形状字符串。

        :return str: 表示多边形几何信息的字符串。
        """
        str_geo = verts_to_str(self.verts)

        str_height = flag_to_str("height", self.height)
        str_nverts = flag_to_str("nverts", self.nverts)

        str_volume_flag = flag_to_str("volume_flag", self.volume_flag)

        str_split_flag = flag_to_str("split_flag", self.split_flag)
        str_plane = flag_to_str("plane", self.plane)
        str_changes = flag_to_str("changes", self.changes)

        flags = [
            str_volume_flag,
            str_split_flag,
            str_changes,
            str_nverts,
            str_plane,
            str_height,
        ]

        str_flags = list_str(flags)
        add_string = "        setval nverts {}\n".format(str(self.nverts))
        shape = self.str_shape(str_flags, str_geo, add_string=add_string)
        self.shape = shape
        return shape
