    def from_cyl(self, cylinder):
        """
        从给定的圆柱对象中提取信息并初始化当前对象。

        :param cylinder: 要从中提取信息的圆柱对象。

        :return self: 返回当前对象，以便进行链式调用。
        """
        self.geometry = cylinder
        self.refresh_point()
        center = cylinder.Center
        radius = cylinder.Radius
        plane = cylinder.BasePlane
        normal = plane.Normal
        height = abs(cylinder.Height2 - cylinder.Height1)
        self.height = height
        self.center1 = center
        if abs(normal[0]) > 0 and normal[1] == 0 and normal[2] == 0:
            self.plane = 0
            center2 = rg.Point3d(center.X + height, center.Y, center.Z)
        elif abs(normal[1]) > 0 and normal[0] == 0 and normal[2] == 0:
            self.plane = 1
            center2 = rg.Point3d(center.X, center.Y + height, center.Z)
        elif abs(normal[2]) > 0 and normal[1] == 0 and normal[0] == 0:
            self.plane = 2
            center2 = rg.Point3d(center.X, center.Y, center.Z + height)

        self.center2 = center2
        self.radius = radius
        self.radius2 = radius
        self.geo_to_str()
        return self

    def geo_to_str(self):
        """
        将圆柱的几何信息转换为字符串表示形式。

        该方法将圆柱的中心点、半径、高度等信息转换为字符串，并将这些字符串拼接成一个完整的形状字符串。

        :return str: 表示圆柱几何信息的字符串。
        """
        str_geo = points_to_str(self.center1, self.center2)

        str_volume_flag = flag_to_str("volume_flag", self.volume_flag)
        str_changes = flag_to_str("changes", self.changes)
        str_plane = flag_to_str("plane", self.plane)
        str_radius = flag_to_str("radius", self.radius)
        str_iradius = flag_to_str("iradius", self.iradius)
        str_radius2 = flag_to_str("radius2", self.radius2)
        str_iradius2 = flag_to_str("iradius2", self.iradius2)
        str_height = flag_to_str("height", self.height)
        str_start_angle = flag_to_str("start_angle", self.start_angle)
        str_end_angle = flag_to_str("end_angle", self.end_angle)

        flags = [
            str_volume_flag,
            str_changes,
            str_plane,
            str_radius,
            str_iradius,
            str_radius2,
            str_iradius2,
            str_height,
            str_start_angle,
            str_end_angle,
        ]

        str_flags = list_str(flags)
        shape = self.str_shape(str_geo, str_flags)
        self.shape = shape
        return shape
