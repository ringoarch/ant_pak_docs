def create_perpendicular_line_and_find_intersections(circle, line_a):
    # 获取圆的圆心
    circle_center = circle.Center

    # 获取直线A的方向向量，并将其归一化
    direction_vector_a = line_a.Direction
    direction_vector_a.Unitize()

    # 计算垂直于直线A的方向向量，这里假设直线A和直线B都在XY平面上
    direction_vector_b = rg.Vector3d(-direction_vector_a.Y, direction_vector_a.X, 0)

    # 创建直线B，使其通过圆心并垂直于直线A
    line_b_start = (
        circle_center - direction_vector_b * circle.Radius * 2
    )  # 乘以2只是为了确保线足够长
    line_b_end = circle_center + direction_vector_b * circle.Radius * 2
    line_b = rg.Line(line_b_start, line_b_end)

    # 计算直线B与圆的交点
    intersection_results = rg.Intersect.Intersection.CurveCurve(
        line_b.ToNurbsCurve(), circle.ToNurbsCurve(), 0.001, 0.001
    )

    # 提取交点
    intersection_points = [result.PointA for result in intersection_results]

    return intersection_points

def create_line_at_angle(
    base_line, negtive_direction=False, angle_degrees=30, line_length=10
):
    # Get the midpoint of the base line
    midpoint = base_line.PointAt(0.5)

    # Default direction vector (assume horizontal direction)
    if negtive_direction:
        direction_vector = rg.Vector3d(1, 0, 0)
        angle_radians = math.radians(angle_degrees)
    else:
        direction_vector = rg.Vector3d(-1, 0, 0)
        angle_radians = -math.radians(angle_degrees)
    # Normalize the vector
    direction_vector.Unitize()

    # Calculate the rotation axis (assume Z-axis for 2D rotation)
    rotation_axis = rg.Vector3d.ZAxis

    # Create the rotation transformation
    rotation_transform = rg.Transform.Rotation(angle_radians, rotation_axis, midpoint)

    # Rotate the direction vector
    rotated_vector = direction_vector
    rotated_vector.Transform(rotation_transform)

    # Scale the vector to the desired length
    rotated_vector *= line_length

    # Create the new end point
    end_point = midpoint + rotated_vector

    # Create the new line
    new_line = rg.Line(midpoint, end_point)
    return new_line

def create_sun(base_line, negtive_direction=False, angle_degrees=30):
    new_line = create_line_at_angle(base_line, negtive_direction, angle_degrees)
    circle = rg.Circle(new_line.To, 1)
    intersection_points = create_perpendicular_line_and_find_intersections(
        circle, new_line
    )
    # Create a polyline (triangle) using the base line and the new end point
    polyline = [base_line.To, base_line.From] + intersection_points
    return circle, polyline

def create_shade(base_line, negtive_direction=False, shade_distance=0, shade_depth=1):
    # Get the midpoint of the base line
    midpoint = sorted([base_line.From, base_line.To], key=lambda point: point.Y)[-1]
    midpoint = move_point(midpoint, rg.Vector3d(0, shade_distance, 0))
    # Default direction vector (assume horizontal direction)
    line_length = shade_depth  # Length of the new line
    if negtive_direction:
        direction_vector = rg.Vector3d(1, 0, 0)

    else:
        direction_vector = rg.Vector3d(-1, 0, 0)
    angle_radians = math.radians(0)
    # Normalize the vector

def move_point(point, vector):
    # point是一个rg.Point3d对象，vector是一个rg.Vector3d对象
    # 直接修改点的坐标
    return rg.Point3d(point.X + vector.X, point.Y + vector.Y, point.Z + vector.Z)

def find_line_intersection(line1, line2):
    # 创建两条直线的交点
    intersection_success, a, b = rg.Intersect.Intersection.LineLine(line1, line2)

    # 如果成功找到交点，并且a和b的值非常接近，表示确实有交点
    if intersection_success:
        # 计算交点的位置
        intersection_point = line1.PointAt(a)
        return intersection_point
    else:
        print_time("Can not find intersection point")
        return None

def sort_points_clockwise(points, base_point):
    # 计算每个点相对于基准点的角度
    def angle_from_base(point):
        return math.atan2(point[1] - base_point[1], point[0] - base_point[0])

    # 排序函数，用角度进行比较
    points.sort(key=lambda point: angle_from_base(point), reverse=True)

    return points

def extend_line_from_start(line, scale_factor):
    # 获取线条的起点和终点
    start_point = line.From
    end_point = line.To

    # 计算线条的方向向量
    direction_vector = end_point - start_point

    # 根据给定的比例因子扩展这个向量
    extended_vector = direction_vector * scale_factor

    # 计算新的终点
    new_end_point = start_point + extended_vector

    # 创建新的线条
    extended_line = rg.Line(start_point, new_end_point)

    return extended_line

def create_curve_arrow(curve, arrow_size=0.2):

    end_point = curve.PointAtEnd
    # Create the arrow shaft
    shaft = curve
    # Calculate the direction vector and normalize it
    direction = curve.TangentAtEnd
    direction.Unitize()
    arrow_thickness = 1.5 * arrow_size
    # Calculate the base point of the arrowhead
    arrow_base = end_point - direction * arrow_size

    # Create the arrowhead
    perp_vector1 = rg.Vector3d.CrossProduct(direction, rg.Vector3d.ZAxis)
    perp_vector1.Unitize()
    perp_vector2 = rg.Vector3d.CrossProduct(direction, perp_vector1)
    perp_vector2.Unitize()

    arrowhead_point1 = arrow_base + perp_vector1 * arrow_thickness
    arrowhead_point2 = arrow_base - perp_vector1 * arrow_thickness

    arrowhead = rg.Polyline([arrowhead_point1, end_point, arrowhead_point2])

    return [shaft, arrowhead]

def create_offset_curve_arrow(
    curve, offset_distance, arrow_tip_width=0.6, arrow_size=0.3, magnitude_factor=0.02
):
    if not isinstance(curve, rg.Curve):
        return None

    def random_vector(magnitude, negative=1):
        return rg.Vector3d(negative * magnitude, negative * magnitude, 0)

    def randomize_points(curve, magnitude_factor=0.02, negative=1):
        nurbs_curve = curve.ToNurbsCurve()
        control_points = nurbs_curve.Points
        magnitude = nurbs_curve.GetLength() * magnitude_factor
        for i in range(2, control_points.Count - 2):
            point = control_points[i].Location
            random_vec = random_vector(magnitude, negative)
            control_points.SetPoint(i, point + random_vec)
        return nurbs_curve

    def reduce_control_points(curve, num_points):
        nurbs_curve = curve.ToNurbsCurve()
        degree = nurbs_curve.Degree
        if num_points < degree + 1:
            num_points = degree + 1
        new_curve = nurbs_curve.Fit(degree, num_points, 0.01)
        return new_curve

    def offset_curve(curve, distance):
        offset_curves = curve.Offset(
            rg.Plane.WorldXY, distance, 0.01, rg.CurveOffsetCornerStyle.Sharp
        )
        if len(offset_curves) == 0:
            return None
        return offset_curves[0]

    # Reduce control points
    curve = reduce_control_points(curve, 3)

    # Calculate dimensions based on curve length
    length = curve.GetLength()
    arrow_tip_width = length * arrow_tip_width * offset_distance / 1.5
    arrow_size = length * arrow_size * offset_distance / 1.5
    offset_distance = offset_distance * length / 20

    # Offset the curve in both directions
    offset_curve1 = offset_curve(curve, offset_distance)
    if not offset_curve1:
        return None
    offset_curve2 = offset_curve(curve, -offset_distance)
    if not offset_curve2:
        return None

    # Randomize control points of the offset curves
    offset_curve1 = randomize_points(
        offset_curve1, negative=-1, magnitude_factor=magnitude_factor
    )
    offset_curve2 = randomize_points(offset_curve2, magnitude_factor=magnitude_factor)

    # Get the end and start points of the original curve
    end_point = curve.PointAtEnd
    # start_point = curve.PointAtStart

    # Calculate the direction vector and normalize it
    direction = -curve.TangentAtEnd
    direction.Unitize()

    # Calculate the base point of the arrowhead
    arrow_base = end_point - direction * arrow_size

    # Create the arrowhead
    perp_vector1 = rg.Vector3d.CrossProduct(direction, rg.Vector3d.ZAxis)
    perp_vector1.Unitize()
    arrowhead_point1 = end_point + perp_vector1 * arrow_tip_width / 2
    arrowhead_point2 = end_point - perp_vector1 * arrow_tip_width / 2

    arrowhead = rg.Polyline(
        [
            offset_curve2.PointAtEnd,
            arrowhead_point1,
            arrow_base,
            arrowhead_point2,
            offset_curve1.PointAtEnd,
        ]
    )
    end_curve = rg.Line(offset_curve1.PointAtStart, offset_curve2.PointAtStart)

    return [offset_curve1, offset_curve2, end_curve, arrowhead]

def create_teardrop_shape(circle_center, radius=0.3, length=1):

    # 创建一个圆
    circle = rg.Circle(circle_center, radius)

    # 获取圆的上半部分作为雨滴的顶部
    upper_half_circle = rg.Arc(circle, -math.pi)

    # 计算下部尖端的控制点
    control_point = move_point(circle_center, rg.Vector3d(0, length, 0))

    # 创建从圆的左侧到控制点再到圆的右侧的Bezier曲线
    points = [
        move_point(circle_center, rg.Vector3d(-radius, 0, 0)),
        control_point,
        move_point(circle_center, rg.Vector3d(radius, 0, 0)),
    ]
    bezier_curve = rg.Curve.CreateInterpolatedCurve(points, 3)

    # 合并上半圆和Bezier曲线形成雨滴形状
    curves = [upper_half_circle.ToNurbsCurve(), bezier_curve]

    teardrop_shape = rg.Curve.JoinCurves(curves, 0.01, True)[0]

    return teardrop_shape
